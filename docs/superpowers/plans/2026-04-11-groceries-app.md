# Groceries App Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a Flutter Android app for household grocery management with voice input (Google Home/IFTTT + in-app mic), real-time shared shopping list, pantry tracking with optimal quantities, configurable restock push notifications, store price comparisons via Trolley.co.uk API, and dark mode throughout.

**Architecture:** Flutter (Android) talks to Firebase Firestore via real-time listeners for shared household state. Firebase Cloud Functions handle the IFTTT webhook (Google Home voice adds), Trolley.co.uk price fetching (scheduled twice daily), and FCM restock nudges (scheduled daily). Firebase Auth (Google Sign-In) is the only auth method.

**Tech Stack:** Flutter 3.x / Dart 3.x, firebase_core, firebase_auth, cloud_firestore, firebase_messaging, google_sign_in, flutter_riverpod, go_router, speech_to_text, fl_chart, share_plus. Cloud Functions: Node.js 18 / TypeScript, firebase-admin, firebase-functions, axios.

  ```dart
  // lib/models/item.dart
  import 'package:cloud_firestore/cloud_firestore.dart';

  enum ItemSource { app, voiceInApp, googleHome }

  extension ItemSourceExt on ItemSource {
    String get value => switch (this) {
      ItemSource.app => 'app',
      ItemSource.voiceInApp => 'voice_in_app',
      ItemSource.googleHome => 'google_home',
    };
    static ItemSource fromString(String s) => switch (s) {
      'voice_in_app' => ItemSource.voiceInApp,
      'google_home' => ItemSource.googleHome,
      _ => ItemSource.app,
    };
  }

  class AddedBy {
    final String? uid;
    final String displayName;
    final ItemSource source;
    const AddedBy({required this.uid, required this.displayName, required this.source});

    Map<String, dynamic> toMap() => {
      'uid': uid,
      'displayName': displayName,
      'source': source.value,
    };

    factory AddedBy.fromMap(Map<String, dynamic> m) => AddedBy(
      uid: m['uid'],
      displayName: m['displayName'] ?? 'Unknown',
      source: ItemSourceExt.fromString(m['source'] ?? 'app'),
    );
  }

  class ShoppingItem {
    final String id;
    final String name;
    final int quantity;
    final String categoryId;
    final List<String> preferredStores;
    final String? pantryItemId;
    final AddedBy addedBy;
    final DateTime addedAt;

    const ShoppingItem({
      required this.id, required this.name, required this.quantity,
      required this.categoryId, required this.preferredStores,
      required this.pantryItemId, required this.addedBy, required this.addedAt,
    });

    Map<String, dynamic> toMap() => {
      'name': name, 'quantity': quantity, 'categoryId': categoryId,
      'preferredStores': preferredStores, 'pantryItemId': pantryItemId,
      'addedBy': addedBy.toMap(),
      'addedAt': Timestamp.fromDate(addedAt),
    };

    factory ShoppingItem.fromFirestore(DocumentSnapshot doc) {
      final d = doc.data() as Map<String, dynamic>;
      return ShoppingItem(
        id: doc.id, name: d['name'], quantity: d['quantity'] ?? 1,
        categoryId: d['categoryId'] ?? 'uncategorised',
        preferredStores: List<String>.from(d['preferredStores'] ?? []),
        pantryItemId: d['pantryItemId'],
        addedBy: AddedBy.fromMap(d['addedBy'] ?? {}),
        addedAt: (d['addedAt'] as Timestamp?)?.toDate() ?? DateTime.now(),
      );
    }
  }
  ```

  ```dart
  class HouseholdService {
    final FirebaseFirestore _db;
    HouseholdService({FirebaseFirestore? db}) : _db = db ?? FirebaseFirestore.instance;

    String _randomToken(int length) {
      const chars = 'abcdefghijklmnopqrstuvwxyz0123456789';
      final rand = Random.secure();
      return List.generate(length, (_) => chars[rand.nextInt(chars.length)]).join();
    }

    Future<String> createHousehold(User user, String name) async {
      final inviteToken = _randomToken(32);
      final webhookToken = _randomToken(32);
      final ref = await _db.collection('households').add({
        'name': name,
        'createdBy': user.uid,
        'createdAt': FieldValue.serverTimestamp(),
        'inviteToken': inviteToken,
      });

      final batch = _db.batch();

      // Add member
      batch.set(_db.doc('households/${ref.id}/members/${user.uid}'), {
        'displayName': user.displayName ?? 'User',
        'email': user.email,
        'joinedAt': FieldValue.serverTimestamp(),
        'webhookToken': webhookToken,
        'fcmToken': null,
      });

      // Seed default stores
      for (final s in _defaultStores) {
        final storeRef = _db.collection('households/${ref.id}/stores').doc();
        batch.set(storeRef, {...s, 'addedBy': user.uid, 'addedAt': FieldValue.serverTimestamp()});
      }

      // Seed default categories
      for (final c in _defaultCategories) {
        final catRef = _db.collection('households/${ref.id}/categories').doc();
        batch.set(catRef, {...c, 'addedBy': user.uid, 'addedAt': FieldValue.serverTimestamp()});
      }

      await batch.commit();
      return ref.id;
    }

    Future<String?> joinByInviteToken(User user, String token) async {
      final snap = await _db.collection('households')
          .where('inviteToken', isEqualTo: token)
          .limit(1)
          .get();
      if (snap.docs.isEmpty) return null;

      final householdId = snap.docs.first.id;
      final webhookToken = _randomToken(32);
      await _db.doc('households/$householdId/members/${user.uid}').set({
        'displayName': user.displayName ?? 'User',
        'email': user.email,
        'joinedAt': FieldValue.serverTimestamp(),
        'webhookToken': webhookToken,
        'fcmToken': null,
      });
      return householdId;
    }

    Stream<Household?> householdStream(String uid) {
      return _db.collection('households')
          .where('members.$uid', isNull: false)
          .limit(1)
          .snapshots()
          .asyncMap((snap) async {
            // Check membership via subcollection instead
            return null;
          });
    }

    // Simpler: look up household where user is a member
    Future<String?> getHouseholdIdForUser(String uid) async {
      // CollectionGroup query to find member doc
      final snap = await _db.collectionGroup('members')
          .where(FieldPath.documentId, isEqualTo: uid)
          .limit(1)
          .get();
      if (snap.docs.isEmpty) return null;
      return snap.docs.first.reference.parent.parent?.id;
    }
  ```

  ```
        allow delete: if false;

        match /members/{uid} {

          allow delete: if false;

        match /items/{itemId} {

        match /stores/{storeId} {
  ```

  ```dart
  // lib/services/items_service.dart
  import 'package:cloud_firestore/cloud_firestore.dart';
  import '../models/item.dart';
  import '../models/pantry_item.dart';

  class ItemsService {
    final FirebaseFirestore _db;
    ItemsService({FirebaseFirestore? db}) : _db = db ?? FirebaseFirestore.instance;

    Stream<List<ShoppingItem>> itemsStream(String householdId) {
      return _db
          .collection('households/$householdId/items')
          .orderBy('addedAt', descending: false)
          .snapshots()
          .map((s) => s.docs.map(ShoppingItem.fromFirestore).toList());
    }

    Future<void> addItem({
      required String householdId,
      required String name,
      required String categoryId,
      required List<String> preferredStores,
      required String? pantryItemId,
      required AddedBy addedBy,
      int quantity = 1,
    }) async {
      await _db.collection('households/$householdId/items').add({
        'name': name, 'quantity': quantity, 'categoryId': categoryId,
        'preferredStores': preferredStores, 'pantryItemId': pantryItemId,
        'addedBy': addedBy.toMap(), 'addedAt': FieldValue.serverTimestamp(),
      });
    }

    Future<void> deleteItem({required String householdId, required String itemId}) async {
      await _db.doc('households/$householdId/items/$itemId').delete();
    }

    /// Check off item. If linked to a pantry entry, resets currentQuantity to optimal.
    Future<void> checkOff({
      required String householdId,
      required ShoppingItem item,
      PantryItem? pantryItem,
    }) async {
      final batch = _db.batch();
      batch.delete(_db.doc('households/$householdId/items/${item.id}'));
      if (pantryItem != null && item.pantryItemId != null) {
        batch.update(
          _db.doc('households/$householdId/pantry/${item.pantryItemId}'),
          {
            'currentQuantity': pantryItem.optimalQuantity,
            'lastPurchasedAt': FieldValue.serverTimestamp(),
          },
        );
      }
      await batch.commit();
    }
  }
  ```

  ```dart
  import 'package:cloud_firestore/cloud_firestore.dart';
  import '../models/pantry_item.dart';

  class PantryService {
    final FirebaseFirestore _db;
    PantryService({FirebaseFirestore? db}) : _db = db ?? FirebaseFirestore.instance;

    Stream<List<PantryItem>> pantryStream(String householdId) {
      return _db
          .collection('households/$householdId/pantry')
          .orderBy('name')
          .snapshots()
          .map((s) => s.docs.map(PantryItem.fromFirestore).toList());
    }

    Future<String> addItem({
      required String householdId,
      required String name,
      required String categoryId,
      required List<String> preferredStores,
      required int optimalQuantity,
      required int currentQuantity,
      int? restockAfterDays,
    }) async {
      final ref = await _db.collection('households/$householdId/pantry').add({
        'name': name, 'categoryId': categoryId, 'preferredStores': preferredStores,
        'optimalQuantity': optimalQuantity, 'currentQuantity': currentQuantity,
        'restockAfterDays': restockAfterDays,
        'lastNudgedAt': null, 'lastPurchasedAt': null,
      });
      return ref.id;
    }

    Future<void> updateItem(String householdId, String itemId, Map<String, dynamic> updates) async {
      await _db.doc('households/$householdId/pantry/$itemId').update(updates);
    }

    Future<void> decrementQuantity({
      required String householdId,
      required String itemId,
      required int current,
    }) async {
      if (current <= 0) return;
      await _db.doc('households/$householdId/pantry/$itemId').update({
        'currentQuantity': current - 1,
      });
    }

    Future<void> incrementQuantity({
      required String householdId,
      required String itemId,
      required int current,
    }) async {
      await _db.doc('households/$householdId/pantry/$itemId').update({
        'currentQuantity': current + 1,
      });
    }

    Future<void> deleteItem(String householdId, String itemId) async {
      await _db.doc('households/$householdId/pantry/$itemId').delete();
    }
  }
  ```

  ```dart
        'addedBy': uid,
        'addedAt': FieldValue.serverTimestamp(),
      });
    }

    Future<void> renameCategory(String householdId, String categoryId, String newName) async {
      await _db.doc('households/$householdId/categories/$categoryId').update({'name': newName});
    }

    /// Deletes category and reassigns all its items to 'uncategorised'.
    Future<void> deleteCategory(String householdId, String categoryId) async {
      final uncatSnap = await _db.collection('households/$householdId/categories')
          .where('name', isEqualTo: 'Uncategorised').limit(1).get();
      final uncatId = uncatSnap.docs.isNotEmpty ? uncatSnap.docs.first.id : 'uncategorised';

      final batch = _db.batch();
      final itemsSnap = await _db.collection('households/$householdId/items')
          .where('categoryId', isEqualTo: categoryId).get();
      for (final doc in itemsSnap.docs) batch.update(doc.reference, {'categoryId': uncatId});
      final pantrySnap = await _db.collection('households/$householdId/pantry')
          .where('categoryId', isEqualTo: categoryId).get();
      for (final doc in pantrySnap.docs) batch.update(doc.reference, {'categoryId': uncatId});
      batch.delete(_db.doc('households/$householdId/categories/$categoryId'));
      await batch.commit();
    }
  }
  ```

  ```dart
  // lib/services/notification_service.dart
  import 'package:cloud_firestore/cloud_firestore.dart';
  import 'package:firebase_messaging/firebase_messaging.dart';
  import 'package:go_router/go_router.dart';
  import 'package:flutter/material.dart';

  class NotificationService {
    final FirebaseMessaging _fcm = FirebaseMessaging.instance;
    final FirebaseFirestore _db;
    NotificationService({FirebaseFirestore? db}) : _db = db ?? FirebaseFirestore.instance;

    Future<void> init(String householdId, String uid, GlobalKey<NavigatorState> navigatorKey) async {
      await _fcm.requestPermission();
      final token = await _fcm.getToken();
      if (token != null) {
        await _db.doc('households/$householdId/members/$uid').update({'fcmToken': token});
      }

      // Handle notification tap when app is terminated or in background
      FirebaseMessaging.onMessageOpenedApp.listen((message) {
        final pantryItemId = message.data['pantryItemId'];
        if (pantryItemId != null) {
          navigatorKey.currentContext?.go('/pantry/$pantryItemId');
        }
      });
    }

    Future<String?> getWebhookToken(String householdId, String uid) async {
      final doc = await _db.doc('households/$householdId/members/$uid').get();
      return (doc.data() as Map<String, dynamic>?)?['webhookToken'];
    }

    Future<void> rotateWebhookToken(String householdId, String uid) async {
      // Generate new 32-char token
      const chars = 'abcdefghijklmnopqrstuvwxyz0123456789';
      final rand = List.generate(32, (_) => chars[DateTime.now().microsecond % chars.length]).join();
      await _db.doc('households/$householdId/members/$uid').update({'webhookToken': rand});
    }
  }
  ```

  ```typescript
  // functions/src/rate_limiter.ts
  import * as admin from "firebase-admin";

  /**
   * Firestore-backed rate limiter (survives cold starts).
   * Allows maxRequests per windowMs per key.
   */
  export async function checkRateLimit(
    token: string,
    maxRequests = 60,
    windowMs = 3_600_000
  ): Promise<boolean> {
    const db = admin.firestore();
    const ref = db.collection("_rate_limits").doc(token);
    const now = Date.now();
    const windowStart = now - windowMs;

    return db.runTransaction(async (tx) => {
      const doc = await tx.get(ref);
      const data = doc.data() ?? {requests: []};
      const recent: number[] = ((data.requests as number[]) ?? []).filter(
        (t) => t > windowStart
      );
      if (recent.length >= maxRequests) return false;
      recent.push(now);
      tx.set(ref, {requests: recent});
      return true;
    });
  }
  ```

  ```typescript
  // functions/src/add_item_webhook.ts
  import * as functions from "firebase-functions";
  import * as admin from "firebase-admin";
  import {checkRateLimit} from "./rate_limiter";

  export const addItemWebhook = functions.https.onRequest(async (req, res) => {
    if (req.method !== "POST") {
      res.status(405).send("Method not allowed");
      return;
    }

    // Token is the last path segment: /addItemWebhook/TOKEN
    const token = req.path.replace(/^\/+/, "").split("/")[0];
    if (!token) {
      res.status(401).send("Unauthorized");
      return;
    }

    // Rate limiting: 60 req/hour per token
    const allowed = await checkRateLimit(token);
    if (!allowed) {
      res.status(429).send("Rate limit exceeded");
      return;
    }

    // Find member by webhook token (collectionGroup query)
    const snap = await admin
      .firestore()
      .collectionGroup("members")
      .where("webhookToken", "==", token)
      .limit(1)
      .get();

    if (snap.empty) {
      res.status(401).send("Unauthorized");
      return;
    }

    const memberDoc = snap.docs[0];
    const householdId = memberDoc.ref.parent.parent?.id;
    if (!householdId) {
      res.status(500).send("Internal error");
      return;
    }

    const {displayName} = memberDoc.data() as {displayName: string};

    // Validate and sanitise item name
    const rawName: unknown = (req.body as Record<string, unknown>)?.item;
    if (!rawName || typeof rawName !== "string") {
      res.status(400).send("Missing item name");
      return;
    }
    const name = rawName.trim().slice(0, 100).replace(/[\x00-\x1F\x7F]/g, "");
    if (!name) {
      res.status(400).send("Invalid item name");
      return;
    }

    await admin
      .firestore()
      .collection(`households/${householdId}/items`)
      .add({
        name,
        quantity: 1,
        categoryId: "uncategorised",
        preferredStores: [],
        pantryItemId: null,
        addedBy: {uid: null, displayName, source: "google_home"},
        addedAt: admin.firestore.FieldValue.serverTimestamp(),
      });

    res.status(200).send("OK");
  });
  ```

- [ ] **Step 1: Write restock_notifications.ts**

  ```typescript
  // functions/src/restock_notifications.ts
  import * as functions from "firebase-functions";
  import * as admin from "firebase-admin";

  interface PantryData {
    name: string;
    currentQuantity: number;
    optimalQuantity: number;
    restockAfterDays: number | null;
    lastNudgedAt: admin.firestore.Timestamp | null;
  }

  export const sendRestockNudges = functions.pubsub
    .schedule("0 8 * * *") // 08:00 UTC daily
    .onRun(async () => {
      const db = admin.firestore();
      const now = admin.firestore.Timestamp.now();
      const msPerDay = 86_400_000;

      const households = await db.collection("households").get();

      for (const householdDoc of households.docs) {
        const householdId = householdDoc.id;

        const members = await db
          .collection(`households/${householdId}/members`)
          .get();
        const tokens: string[] = members.docs
          .map((d) => (d.data() as {fcmToken?: string}).fcmToken)
          .filter((t): t is string => Boolean(t));

        if (tokens.length === 0) continue;

        const pantryItems = await db
          .collection(`households/${householdId}/pantry`)
          .get();

        for (const pantryDoc of pantryItems.docs) {
          const data = pantryDoc.data() as PantryData;
          if (!data.restockAfterDays) continue;

          const lastNudgedMs = data.lastNudgedAt?.toMillis() ?? 0;
          const daysSince =
            (now.toMillis() - lastNudgedMs) / msPerDay;

          if (daysSince < data.restockAfterDays) continue;

          try {
            await admin.messaging().sendEachForMulticast({
              tokens,
              notification: {
                title: "Restock check",
                body: `Did you use any ${data.name}? (${data.currentQuantity} on hand, optimal: ${data.optimalQuantity})`,
              },
              data: {
                type: "restock_nudge",
                householdId,
                pantryItemId: pantryDoc.id,
              },
              android: {
                notification: {channelId: "restock_nudges"},
              },
            });
          } catch (e) {
            functions.logger.error(`Failed to send nudge for ${data.name}`, e);
          }

          await pantryDoc.ref.update({lastNudgedAt: now});
        }
      }
    });
  ```

  ```typescript
  // functions/src/fetch_prices.ts
  import * as functions from "firebase-functions";
  import * as admin from "firebase-admin";
  import axios from "axios";

  const TROLLEY_STORES = [
    "tesco", "asda", "sainsburys", "morrisons",
    "waitrose", "aldi", "lidl", "ocado",
  ];

  interface TrolleyProduct {
    store: string;
    price: number;
  }

  export const fetchPrices = functions.pubsub
    .schedule("0 8,20 * * *") // 08:00 and 20:00 UTC
    .onRun(async () => {
      const db = admin.firestore();
      const apiKey: string = functions.config().trolley?.api_key ?? "";
      if (!apiKey) {
        functions.logger.error("Missing Trolley API key");
        return;
      }

      const households = await db.collection("households").get();

      for (const householdDoc of households.docs) {
        const householdId = householdDoc.id;
        const pantryItems = await db
          .collection(`households/${householdId}/pantry`)
          .get();

        for (const pantryDoc of pantryItems.docs) {
          const {name} = pantryDoc.data() as {name: string};
          const prices: Record<string, number | null> = {};

          try {
            const response = await axios.get<{products: TrolleyProduct[]}>(
              `https://api.trolley.co.uk/search/?q=${encodeURIComponent(name)}&limit=5`,
              {
                headers: {"x-api-key": apiKey},
                timeout: 8000,
              }
            );

            const products = response.data?.products ?? [];
            for (const store of TROLLEY_STORES) {
              const match = products.find((p) => p.store === store);
              prices[store] = match?.price ?? null;
            }
          } catch (e) {
            functions.logger.warn(`Price fetch failed for ${name}`, e);
            for (const store of TROLLEY_STORES) prices[store] = null;
          }

          const snapshotId = Date.now().toString();
          await db
            .collection(
              `households/${householdId}/prices/${pantryDoc.id}/snapshots`
            )
            .doc(snapshotId)
            .set({
              ...prices,
              fetchedAt: admin.firestore.FieldValue.serverTimestamp(),
            });
        }
      }
    });
  ```

  ```dart
  // lib/screens/prices/widgets/price_panel.dart
  import 'package:flutter/material.dart';
  import 'package:flutter_riverpod/flutter_riverpod.dart';
  import '../../../models/pantry_item.dart';
  import '../../../models/price_snapshot.dart';
  import '../../../providers/prices_provider.dart';
  import 'sparkline_chart.dart';

  class PricePanel extends ConsumerWidget {
    final PantryItem item;
    const PricePanel({super.key, required this.item});

    @override
    Widget build(BuildContext context, WidgetRef ref) {
      final snapshotsAsync = ref.watch(pricesProvider(item.id));

      return snapshotsAsync.when(
        loading: () => const LinearProgressIndicator(),
        error: (_, __) => const Text('Price data unavailable'),
        data: (snapshots) {
          if (snapshots.isEmpty) return const Padding(
            padding: EdgeInsets.all(8),
            child: Text('No price data yet — check back after the next fetch.'),
          );

          final latest = snapshots.last;
          final stores = ['tesco', 'asda', 'sainsburys', 'morrisons', 'waitrose', 'aldi', 'lidl', 'ocado'];

          double? minPrice;
          String? cheapestStore;
          for (final store in stores) {
            final price = latest.prices[store];
            if (price != null && (minPrice == null || price < minPrice)) {
              minPrice = price;
              cheapestStore = store;
            }
          }

          // Determine 30-day low for each store
          final thirtyDayLows = <String, double>{};
          for (final store in stores) {
            final prices = snapshots
                .map((s) => s.prices[store])
                .whereType<double>()
                .toList();
            if (prices.isNotEmpty) thirtyDayLows[store] = prices.reduce((a, b) => a < b ? a : b);
          }

          return Column(
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              Text(item.name, style: Theme.of(context).textTheme.titleMedium),
              const SizedBox(height: 8),
              ...stores.map((store) {
                final price = latest.prices[store];
                final isCheapest = store == cheapestStore;
                final isAtLow = price != null && thirtyDayLows[store] != null &&
                    (price - thirtyDayLows[store]!).abs() < 0.01;
                return ListTile(
                  dense: true,
                  title: Row(children: [
                    Text(store, style: TextStyle(
                      fontWeight: isCheapest ? FontWeight.bold : null,
                      color: isCheapest ? Theme.of(context).colorScheme.primary : null,
                    )),
                    if (isAtLow) ...[
                      const SizedBox(width: 4),
                      Icon(Icons.trending_down, size: 14,
                          color: Theme.of(context).colorScheme.primary),
                    ],
                  ]),
                  trailing: price != null
                      ? Text('£${price.toStringAsFixed(2)}')
                      : const Text('–'),
                  subtitle: SparklineChart(snapshots: snapshots, store: store),
                );
              }),
              const Divider(),
            ],
          );
        },
      );
    }
  }
  ```
